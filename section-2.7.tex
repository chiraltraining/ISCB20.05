\begin{frame}[t]{Kurtosis}
\vfill
\centering
\textbf{Kurtosis:} 
Measure of how often extreme values (on either side of
the mean) occur
\end{frame}


% Slide-n
\begin{frame}[t]{frame title}
	\begin{itemize}
		\item Normally distributed data: kurtosis = 3
		\item Excess kurtosis = kurtosis - 3
	\end{itemize}
\end{frame}
